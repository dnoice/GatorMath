Vector Mathematics
import { Vector3 } from 'gatormath';

const a = new Vector3(1, 0, 0);
const b = new Vector3(0, 1, 0);

// Dot product
const dot = a.dot(b);

// Cross product
const cross = a.cross(b);

// Normalize
const normalized = a.normalize();
Matrix Transformations
import { Matrix4 } from 'gatormath';

const transform = new Matrix4()
  .translate(10, 0, 0)
  .rotateZ(Math.PI / 4)
  .scale(2, 2, 2);

// Apply to vector
const result = transform.apply(vector);
Bezier Curves
import { CubicBezier } from 'gatormath';

const curve = new CubicBezier(
  { x: 0, y: 0 },
  { x: 1, y: 2 },
  { x: 3, y: 2 },
  { x: 4, y: 0 }
);

// Get point at t=0.5
const point = curve.at(0.5);
